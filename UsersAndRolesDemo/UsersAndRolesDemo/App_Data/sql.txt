CREATE TABLE Property(
	Id int IDENTITY(1,1) PRIMARY KEY,
	UserId nvarchar(128) NOT NULL,
	title varchar(50),
	summary varchar(MAX),
	propertyType varchar(15),
	numBedrooms int,
	numWashrooms int,
	kitchen int,
	baseRate money,
	address varchar(256),
	builtYear varchar(4),
	smokingAllowed varchar(3),
	maxNumberGuests varchar(2),
	availableDates date,
	dimensions varchar(128),
	FOREIGN KEY(UserId) REFERENCES AspNetUsers(Id)
);

GO

CREATE TABLE PropertyImage(
	Id int IDENTITY(1,1) PRIMARY KEY,
	PropertyId int NOT NULL,
	main nvarchar(MAX),
	bedroom nvarchar(MAX),
	livingroom nvarchar(MAX),
	bathroom nvarchar(MAX),
	kitchen nvarchar(MAX),
	FOREIGN KEY(PropertyId) REFERENCES Property(Id)
);

GO

CREATE TABLE Reservation(
	Id int IDENTITY(1,1) PRIMARY KEY,
	UserId nvarchar(128) NOT NULL,
	PropertyId int NOT NULL,
	startDate Date,
	endDate Date,
	numberGuests varchar(2),
	status varchar(15),
	totalAmount money,
	FOREIGN KEY(UserId) REFERENCES AspNetUsers(Id),
	FOREIGN KEY(PropertyId) REFERENCES Property(Id)
);
